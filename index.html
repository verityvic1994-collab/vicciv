document.getElementById('searchBtn').addEventListener('click', function() { const word = document.getElementById('wordInput').value.trim(); if (word === "") { alert("請先輸入單字！"); return; } // 呼叫免費的字典 API const apiUrl = `https://api.dictionaryapi.dev/api/v2/entries/en/${word}`; fetch(apiUrl) .then(response => { if (!response.ok) { throw new Error("找不到這個單字，請檢查拼字是否正確！"); } return response.json(); }) .then(data => { // 成功抓到資料後，開始拆解並塞入網頁 const wordData = data[0]; document.getElementById('resultWord').innerText = wordData.word; document.getElementById('resultPhonetic').innerText = wordData.phonetic || "暫無發音資訊"; // 清空舊的定義內容 const meaningsContainer = document.getElementById('definitions'); meaningsContainer.innerHTML = ""; // 跑迴圈把所有的詞性和英文解釋撈出來 wordData.meanings.forEach(meaning => { const partOfSpeech = meaning.partOfSpeech; // 詞性 (如 noun, verb) const definitionText = meaning.definitions[0].definition; // 第一個定義 // 建立 HTML 結構 const p = document.createElement('p'); p.innerHTML = `[${partOfSpeech}] ${definitionText}`; meaningsContainer.appendChild(p); }); // 顯示結果區塊 document.getElementById('resultContainer').style.display = "block"; }) .catch(error => { alert(error.message); }); });
